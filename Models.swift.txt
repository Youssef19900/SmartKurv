import Foundation

struct ProductVariant: Hashable, Identifiable, Codable {
    var id: String { "\(unit)-\(organic ? "eco" : "conv")" }
    let unit: String            // "stk", "bundt", "kg", "ltr" m.fl.
    let organic: Bool           // Ã¸kologisk
}

struct Product: Identifiable, Hashable, Codable {
    let id: String
    let name: String
    let variants: [ProductVariant]
}

struct ListItem: Identifiable, Hashable, Codable {
    let id = UUID()
    let product: Product
    let variant: ProductVariant
    var qty: Int
}

struct ShoppingList: Identifiable, Codable {
    let id: UUID
    let createdAt: Date
    var items: [ListItem]

    init(id: UUID = UUID(), createdAt: Date = Date(), items: [ListItem] = []) {
        self.id = id
        self.createdAt = createdAt
        self.items = items
    }
}
